// 基礎顏色
$primary: #FF3333;           // 主色：鮮艷紅色（根據 Reddoor 標誌和英雄區背景）

$white: #FFFFFF;             // 白色：用於文字和背景對比

$border-disabled: #EBEBEB;   // 邊框/禁用狀態：淺灰，保持一致性

// 對比顏色
$heading: #FFFFFF;           // 主標題：白色（在紅色或深灰背景上）
$body-text: #B0B0B0;         // 全站文字/次標題：中灰色（用於深灰或紅色背景）
